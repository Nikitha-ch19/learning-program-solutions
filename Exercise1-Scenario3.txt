SET SERVEROUTPUT ON;

DECLARE
   CURSOR due_loans IS
      SELECT l.LoanID, l.EndDate, c.Name, c.CustomerID
      FROM Loans l
      JOIN Customers c ON l.CustomerID = c.CustomerID
      WHERE l.EndDate BETWEEN SYSDATE AND SYSDATE + 30;

   v_count NUMBER := 0;
BEGIN
   FOR rec IN due_loans LOOP
      DBMS_OUTPUT.PUT_LINE(
         'ðŸ”” Reminder: LoanID ' || rec.LoanID ||
         ' for customer "' || rec.Name || '" (CustomerID: ' || rec.CustomerID || ')' ||
         ' is due on ' || TO_CHAR(rec.EndDate, 'DD-MON-YYYY')
      );
      v_count := v_count + 1;
   END LOOP;

   IF v_count = 0 THEN
      DBMS_OUTPUT.PUT_LINE('âœ… No loans due in the next 30 days.');
   END IF;
END;
